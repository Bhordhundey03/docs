---
title: "List businesses"
description: "Retrieve all businesses registered under your client account."
---

## Endpoint

```
GET /clients/business
```

## Headers

<ParamField header="x-api-key" type="string" required>
  Your public API key.
</ParamField>

<ParamField header="x-api-secret" type="string" required>
  Your API secret. Use a `stl_test_...` value in sandbox, `stl_live_...` in production.
</ParamField>

<ParamField header="businessId" type="string" required>
  Your business ID.
</ParamField>

## Query parameters

<ParamField query="page" type="integer">
  Page number. Defaults to `1`.
</ParamField>

<ParamField query="limit" type="integer">
  Records per page. Defaults to `20`.
</ParamField>

See [Pagination](/essentials/pagination) for full usage.

## Response

<ResponseField name="status" type="boolean">
  `true` on success.
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable outcome.
</ResponseField>

<ResponseField name="data" type="array">
  Array of business objects.
</ResponseField>

<ResponseField name="data[].id" type="string">
  Unique business ID.
</ResponseField>

<ResponseField name="data[].businessName" type="string">
  The business's registered name.
</ResponseField>

<ResponseField name="data[].bvn" type="string">
  Bank Verification Number associated with the business.
</ResponseField>

<ResponseField name="data[].photoIdUrl" type="string">
  URL to the uploaded photo ID document.
</ResponseField>

<ResponseField name="data[].businessDocumentsUrls" type="array">
  Array of URLs to uploaded business documents.
</ResponseField>

<ResponseField name="data[].isApproved" type="boolean">
  `true` if the business has been approved by Interstellas.
</ResponseField>

<ResponseField name="data[].residentialAddress" type="string">
  The registered residential address.
</ResponseField>

<ResponseField name="data[].stateId" type="string">
  ID of the state where the business is located.
</ResponseField>

<ResponseField name="data[].state" type="object">
  State object containing `id` and `name`.
</ResponseField>

<ResponseField name="data[].clientId" type="string">
  The client account this business belongs to.
</ResponseField>

<ResponseField name="pagination.page" type="integer">
  The current page number.
</ResponseField>

<ResponseField name="pagination.limit" type="integer">
  The page size used for this response.
</ResponseField>

<ResponseField name="pagination.total" type="integer">
  Total number of records matching the current query, across all pages.
</ResponseField>

<ResponseField name="pagination.totalPages" type="integer">
  The final page number, derived from `total ÷ limit`. See [Pagination](/essentials/pagination) — this endpoint uses Shape B.
</ResponseField>
